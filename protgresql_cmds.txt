1. psql.exe -U postgres [-d contact_book] -W   (DB should already be present)

2. to connect to DB: \connect contact_book;    (if not specified in Step 1)
    
3. create database contact_book;

4. create table users
(
    id         bigserial
        constraint users_pk primary key,
first_name varchar(100),
last_name  varchar(100),
contact    bigint,
mail       varchar(200)
)


5. create table auth_user
(
    id         bigserial,
userid     bigint not null unique constraint id references users,
user_name  varchar(50) not null unique,
password   text
);

6. alter table auth_user add constraint auth_user_pk primary key (id, userid, user_name);

7. insert into users values (1, 'Shreya', 'Turior', 9955052050,'shr@gmail.com');

8. \d users  -----> describe the table

9. alter table auth_user drop constraint auth_user_pk;  ---> to drop added constraint

10. alter table auth_user add constraint auth_user_pk primary key(id,userid,user_name);

11.\h ----> help
